# language-separate
separate language by js
<script>
              var type = navigator.appName
							var lang;
							if (type=="Netscape")
								lang = navigator.language
							else
								lang = navigator.userLanguage
							

							var lang = lang.substr(0,2)
							

							if (lang == "ko")
								window.location.replace('')

							else
								window.location.replace('')
</script>
